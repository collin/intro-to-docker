```go server.go
package main

import (
    "fmt"
    "net/http"
)

func helloHandler(w http.ResponseWriter, r *http.Request) {
    fmt.Fprintf(w, "Hello, world!")
}

func main() {
    http.HandleFunc("/", helloHandler)
    http.ListenAndServe(":80", nil)
}
```

```dockerfile Dockerfile
FROM go

COPY server.go
RUN go build -o /server server.go
```
